145 task axi_monitor::monitor_write_data() {
146  axi_seq_item_w_vector_s w_s;
147  axi_seq_item item=null;
148  axi_seq_item cloned_item=null;
149 
150  if (m_config.drv_type != axi_uvm_pkg::e_RESPONDER) {
151  return;
152  }
153 
154  forever {
155  uvm_info(this.get_type_name(),
156  "========> wait_for_write_data()",
157  UVM_HIGH)
158 
159  vif.wait_for_write_data(.s(w_s));
160  uvm_info(this.get_type_name(), "wait_for_write_data - DONE", UVM_HIGH)
161 
162  // Can we just queue the data no matter what and
163  // if the addresshasn'tarrived, we don't sit and poll continuosly
164  // for and address.
165  // What happens if we don't get an address until after wlast?
166  w_q.push_back(w_s);
167 
168  if (item == null) {
169  if (aw_mbx.num() > 0) {
170  aw_mbx.get(item);
171  $cast(cloned_item, item.clone());
172  cloned_item.cmd=e_WRITE_DATA;
173  cloned_item.initialize();
174  cloned_item.dataoffset=0;
175 
176  //
177  cloned_item.wstrb = new[cloned_item.len];
178  cloned_item.data = new[cloned_item.len];
179  }
180  }
181 
182  // if anything in data queue, write it out
183  if (item != null) {
184  while (w_q.size() > 0) {
185 
186  w_s=w_q.pop_front();
187  uvm_info(this.get_type_name(),
188  $sformatf("Lower_Byte_Lane=%0d, Upper_Byte_Lane=%0d, offset=%0d",
189  cloned_item.Lower_Byte_Lane, cloned_item.Upper_Byte_Lane,
190  cloned_item.dataoffset),
191  UVM_HIGH)
192  for (int i=cloned_item.Lower_Byte_Lane;i <=cloned_item.Upper_Byte_Lane;i++) {
193  // wstrb may not be asserted. check
194  if (w_s.wstrb[i]==0b1) {
195  m_memory.write(cloned_item.Start_Address+cloned_item.dataoffset,
196  w_s.wdata[i*8+:8]);
197  cloned_item.data[cloned_item.dataoffset]=w_s.wdata[i*8+:8];
198  }
199  // record wstrb as well so anything else that
200  // needs or wants to fiddle with data, can.
201  cloned_item.wstrb[cloned_item.dataoffset]=w_s.wstrb[i];
202  cloned_item.dataoffset++;
203  }
204  cloned_item.update_address();
205  if (w_s.wlast == 0b1) { // @Todo: count, dont rely on wlast?
206  ap.write(cloned_item);
207  item=null;
208  }
209  } // while
210  }// if
211  } // forever
212 }
