170 
171  int item_needs_init=1;
172 
173  vif.set_awvalid(0b0);
174 
175  forever {
176 
177  if (item == null) {
178  writeaddress_mbx.get(item);
179  uvm_info("axi_driver::write_address",
180  $sformatf("Item: %s", item.convert2string()),
181  UVM_HIGH)
182  item_needs_init=1;
183  }
184 
185  vif.wait_for_clks(.cnt(1));
186 
187  // if done with this xfer (write address is only one clock, done with valid & ready
188  if (vif.get_awready_awvalid == 0b1) {
189  writedata_mbx.put(item);
190  item=null;
191 
192  minval=min_clks_between_aw_transfers;
193  maxval=max_clks_between_aw_transfers;
194  wait_clks_before_next_aw=$urandom_range(maxval,minval);
195 
196  // Check if delay wanted
197  if (wait_clks_before_next_aw==0) {
198  // if not, check if there's another item
199  writeaddress_mbx.try_get(item);
200  if (item!=null) {
201  item_needs_init=1;
202  }
203  }
204  }
205  // Initialize values
206  if (item_needs_init==1) {
207  axi_uvm_pkg::aw_from_class(.t(item), .v(v));
208 
209  item_needs_init=0;
210  }
211 
212  // Update values <- No need in write address (only one clk per)
213 
214  // Write out
215  if (item != null) {
216  vif.write_aw(.s(v), .valid(0b1));
217  if (wait_clks_before_next_aw > 0) {
218  vif.wait_for_clks(.cnt(wait_clks_before_next_aw-1)); // -1 because another wait
219  // at beginning of loop
220  }
221  } // if (item != null)
222 
223  // No item for next clock, so close out bus
224  if (item == null) {
225  v.awaddr = 0x0;
226  v.awid = 0x0;
227  v.awsize = 0x0;
228  v.awburst = 0x0;
229  vif.write_aw(.s(v), .valid(0b0));
230  vif.wait_for_clks(.cnt(1));
231  }
232 
233  } // forever
