157 task axi_driver::write_address() {
158 
159  axi_seq_item item=null;
160  axi_seq_item_aw_vector_s v;
161 
162  bit <63:0> aligned_addr;
163 
164  int minval;
165  int maxval;
166  int wait_clks_before_next_aw;
167 
168  int item_needs_init=1;
169 
170  vif.set_awvalid(0b0);
171 
172  forever {
173 
174  if (item == null) {
175  writeaddress_mbx.get(item);
176  uvm_info("axi_driver::write_address",
177  $sformatf("Item: %s", item.convert2string()),
178  UVM_INFO)
179  item_needs_init=1;
180  }
181 
182  vif.wait_for_clks(.cnt(1));
183 
184  // if done with this xfer (write address is only one clock, done with valid & ready
185  if (vif.get_awready_awvalid == 0b1) {
186  writedata_mbx.put(item);
187  item=null;
188 
189  minval=min_clks_between_aw_transfers;
190  maxval=max_clks_between_aw_transfers;
191  wait_clks_before_next_aw=$urandom_range(maxval,minval);
192 
193  // Check if delay wanted
194  if (wait_clks_before_next_aw==0) {
195  // if not, check if there's another item
196  writeaddress_mbx.try_get(item);
197  if (item!=null) {
198  item_needs_init=1;
199  }
200  }
201  }
202  // Initialize values
203  if (item_needs_init==1) {
204  item.aw_from_class(.t(item), .v(v));
205 
206  item_needs_init=0;
207  }
208 
209  // Update values <- No need in write address (only one clk per)
210 
211  // Write out
212  if (item != null) {
213  vif.write_aw(.s(v), .valid(0b1));
214  if (wait_clks_before_next_aw > 0) {
215  vif.wait_for_clks(.cnt(wait_clks_before_next_aw-1)); // -1 because another wait
216  // at beginning of loop
217  }
218  } // if (item != null)
219 
220  // No item for next clock, so close out bus
221  if (item == null) {
222  v.awaddr = 0x0;
223  v.awid = 0x0;
224  v.awsize = 0x0;
225  v.awburst = 0x0;
226  vif.write_aw(.s(v), .valid(0b0));
227  vif.wait_for_clks(.cnt(1));
228  }
229 
230  } // forever

247 task axi_driver::write_data() {
248  axi_seq_item item=null;
249  axi_seq_item_w_vector_s s;
250 
251  bit iaxi_incompatible_wready_toggling_mode;
252 
253  int n=0;
254 
255  int minval;
256  int maxval;
257  int wait_clks_before_next_w;
258 
259  vif.set_wvalid(0b0);
260  forever {
261 
262  if (item == null) {
263  writedata_mbx.get(item);
264  item.initialize();
265  uvm_info("axi_driver::write_data",
266  $sformatf("Item: %s", item.convert2string()),
267  UVM_INFO)
268  }
269 
270  // Look at this only one per loop, so there's no race condition of it
271  // changing mid-loop.
272  iaxi_incompatible_wready_toggling_mode = axi_incompatible_wready_toggling_mode;
273 
274  vif.wait_for_clks(.cnt(1));
275 
276  // defaults. not needed but I think is cleaner in sim
277  s.wvalid = 0b0;
278  s.wdata = 0xfeed_beef;
279  s.wstrb = 0x0;
280  s.wlast = 0b0;
281 
282  // Check if done with this transfer
283  if (vif.get_wready()==0b1 && vif.get_wvalid() == 10b1) {
284  item.dataoffset = n;
285  if (iaxi_incompatible_wready_toggling_mode == 0b0) {
286  item.validcntr++;
287  }
288 
289  item.update_address();
290 
291  if (item.dataoffset>=item.Burst_Length_Bytes) { //F
292  writeresponse_mbx.put(item);
293  item = null;
294 
295  minval=min_clks_between_w_transfers;
296  maxval=max_clks_between_w_transfers;
297  wait_clks_before_next_w=$urandom_range(maxval,minval);
298 
299  // Check if delay wanted
300  if (wait_clks_before_next_w==0) {
301  // if not, check if there's another item
302  writedata_mbx.try_get(item);
303 
304  if (item != null) {
305  item.initialize();
306  }
307  }
308  }
309  } // (vif.get_wready()==1'b1 && vif.get_wvalid() == 1'b1)
310 
311 
312  // Update values
313  if (item != null) {
314 
315  if (item.validcntr >= item.validcntr_max) {
316  item.validcntr=0;
317  }
318 
319  //
320  // if invalid-toggling-mode is enabled, then allow deasserting valid
321  // before ready asserts.
322  // Default is to stay asserted, and only allow deasssertion after ready asserts.
323  if (iaxi_incompatible_wready_toggling_mode == 0b0) {
324  if (vif.get_wvalid() == 0b0) {
325  item.validcntr++;
326  }
327  } else {
328  item.validcntr++;
329  }
330 
331  s.wvalid = item.valid[item.validcntr]; // 1'b1;
332  s.wstrb = 0x0;
333  s.wdata = 0x0;
334  s.wlast = 0b0;
335  n=item.dataoffset;
336  for (int j=item.Lower_Byte_Lane;j <=item.Upper_Byte_Lane;j++) {
337  s.wdata[j*8+:8] = item.data[n++];
338  s.wstrb[j] = 0b1;
339  if (n>=item.Burst_Length_Bytes) {
340  s.wlast=0b1;
341  break;
342  }
343  } // for
344 
345  // Write out
346  vif.write_w(.s(s));
347 
348 
349  } // (item != null)
350 
351  // No item for next clock, so close out bus
352  if (item == null) {
353  s.wvalid = 0b0;
354  s.wlast = 0b0;
355  s.wdata = 0x0;
356  // s.wid = 'h0; AXI3 only
357  s.wstrb = 0x0;
358 
359  vif.write_w(.s(s));
360 
361  if (wait_clks_before_next_w > 0) {
362  vif.wait_for_clks(.cnt(wait_clks_before_next_w-1));
363  // -1 because another wait
364  // at beginning of loop
365  }
366  } // if (item == null
367  } // forever
