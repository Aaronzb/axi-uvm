334 task axi_driver::driver_write_data() {
335  axi_seq_item item=null;
336  axi_seq_item_w_vector_s s;
337 
338  bit iaxi_incompatible_wready_toggling_mode;
339 
340  int n=0;
341 
342  int minval;
343  int maxval;
344  int wait_clks_before_next_w;
345 
346  vif.set_wvalid(0b0);
347  forever {
348 
349  if (item == null) {
350  driver_writedata_mbx.get(item);
351  item.initialize();
352  }
353 
354  // Look at this only one per loop, so there's no race condition of it
355  // changing mid-loop.
356  iaxi_incompatible_wready_toggling_mode = axi_incompatible_wready_toggling_mode;
357 
358  vif.wait_for_clks(.cnt(1));
359 
360  // defaults. not needed but I think is cleaner in sim
361  s.wvalid = 0b0;
362  s.wdata = 0xfeed_beef;
363  s.wstrb = 0x0;
364  s.wlast = 0b0;
365 
366  // Check if done with this transfer
367  if (vif.get_wready()==0b1 && vif.get_wvalid() == 10b1) {
368  item.dataoffset = n;
369  if (iaxi_incompatible_wready_toggling_mode == 0b0) {
370  item.validcntr++;
371  }
372 
373  item.update_address();
374 
375  if (item.dataoffset>=item.Burst_Length_Bytes) { //F
376  driver_writeresponse_mbx.put(item);
377  item = null;
378 
379  minval=min_clks_between_w_transfers;
380  maxval=max_clks_between_w_transfers;
381  wait_clks_before_next_w=$urandom_range(maxval,minval);
382 
383  // Check if delay wanted
384  if (wait_clks_before_next_w==0) {
385  // if not, check if there's another item
386  driver_writedata_mbx.try_get(item);
387 
388  if (item != null) {
389  item.initialize();
390  }
391  }
392  }
393  } // (vif.get_wready()==1'b1 && vif.get_wvalid() == 1'b1)
394 
395 
396  // Update values
397  if (item != null) {
398 
399  if (item.validcntr >= item.validcntr_max) {
400  item.validcntr=0;
401  }
402 
403  //
404  // if invalid-toggling-mode is enabled, then allow deasserting valid
405  // before ready asserts.
406  // Default is to stay asserted, and only allow deasssertion after ready asserts.
407  if (iaxi_incompatible_wready_toggling_mode == 0b0) {
408  if (vif.get_wvalid() == 0b0) {
409  item.validcntr++;
410  }
411  } else {
412  item.validcntr++;
413  }
414 
415  s.wvalid = item.valid[item.validcntr]; // 1'b1;
416  s.wstrb = 0x0;
417  s.wdata = 0x0;
418  s.wlast = 0b0;
419  n=item.dataoffset;
420  for (int j=item.Lower_Byte_Lane;j <=item.Upper_Byte_Lane;j++) {
421  s.wdata[j*8+:8] = item.data[n++];
422  s.wstrb[j] = 0b1;
423  if (n>=item.Burst_Length_Bytes) {
424  s.wlast=0b1;
425  break;
426  }
427  } // for
428 
429  // Write out
430  vif.write_w(.s(s));
431 
432 
433  } // (item != null)
434 
435  // No item for next clock, so close out bus
436  if (item == null) {
437  s.wvalid = 0b0;
438  s.wlast = 0b0;
439  s.wdata = 0x0;
440  // s.wid = 'h0; AXI3 only
441  s.wstrb = 0x0;
442 
443  vif.write_w(.s(s));
444 
445  if (wait_clks_before_next_w > 0) {
446  vif.wait_for_clks(.cnt(wait_clks_before_next_w-1));
447  // -1 because another wait
448  // at beginning of loop
449  }
450  } // if (item == null
451  } // forever
