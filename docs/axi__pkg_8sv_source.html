253  bit <C_AXI_LEN_WIDTH-1:0> calculate_axlen(
254  input bit <C_AXI_ADDR_WIDTH-1:0> addr,
255  input bit <2:0> burst_size,
256  input shortint burst_length) {
257 
258 
259  byte unalignment_offset;

261  shortint shifter;
262  shortint ishifter;
263  bit <C_AXI_LEN_WIDTH-1:0> beats;
264 
265  string msg_s;
266 
267  case (burst_size) {
268  e_1BYTE : unalignment_offset = 0;
269  e_2BYTES : unalignment_offset = byte(addr[0]);
270  e_4BYTES : unalignment_offset = byte(addr[2]);
271  e_8BYTES : unalignment_offset = byte(addr[3]);
272  e_16BYTES : unalignment_offset = byte(addr[4]);
273  e_32BYTES : unalignment_offset = byte(addr[5]);
274  e_64BYTES : unalignment_offset = byte(addr[6]);
275  e_128BYTES : unalignment_offset = byte(addr[7]);
276  }
277 
278  total_length=burst_length + unalignment_offset;
279 
280  shifter = shortint(total_length/(2**burst_size));
281 
282  ishifter = shifter*(2**burst_size);
283 
284  if (ishifter != total_length) {
285  shifter += 1;
286  }
287 
288  beats = shifter - 1;
289 
290 
291  msg_s="";
292  $sformat(msg_s, "%s addr: 0x%0x", msg_s, addr);
293  $sformat(msg_s, "%s burst_size: %0d", msg_s, burst_size);
294  $sformat(msg_s, "%s unalignment_offset: %0d", msg_s, unalignment_offset);
295  $sformat(msg_s, "%s burst_length: %0d", msg_s, burst_length);
296  $sformat(msg_s, "%s total_length: %0d", msg_s, total_length);
297  $sformat(msg_s, "%s shifter: %0d", msg_s, shifter);
298  $sformat(msg_s, "%s ishifter: %0d", msg_s, ishifter);
299 
300  uvm_info("axi_pkg::calculate_beats",
301  msg_s,
302  UVM_HIGH)
303 
304  return beats;
